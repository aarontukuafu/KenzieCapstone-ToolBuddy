22      * Returns input string with environment variable references expanded, e.g. $SOME_VAR or ${SOME_VAR}
23      *
24      * Gratefully adapted from Tim Lewis's answer on
25      * https://stackoverflow.com/questions/2263929/regarding-application-properties-file-and-environment-variable
26      */
27     private static String resolveEnvVars(Map<String,String> envVars, String input, boolean failSilently) {
28         if (null == input) {
29             return null;
30         }
31         // match ${ENV_VAR_NAME} or $ENV_VAR_NAME
32         Pattern p = Pattern.compile("\\$\\{(\\w+)\\}|\\$(\\w+)");
33         Matcher m = p.matcher(input); // get a matcher object
34         StringBuffer sb = new StringBuffer();
35         while (m.find()) {
36             String envVarName = null == m.group(1) ? m.group(2) : m.group(1);
37             String envVarValue = envVars.get(envVarName);
38 
39             if(failSilently) {
40                 //how the shell works natively
41                 m.appendReplacement(sb, null == envVarValue ? "" : envVarValue);
42             }else {
43                 //making it throw errors, so we know when one is not found
44                 if (envVarValue == null) {
45                     throw new RuntimeException("Environment variable " + envVarName +
46                             " expected but not found or is null. Please set variable to a non-null value.");
47                 }
48                 m.appendReplacement(sb, envVarValue);
49             }
50         }
51         m.appendTail(sb);
52         return sb.toString();
53     }
